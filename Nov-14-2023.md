Write function RemoveExclamationMarks which removes all exclamation marks from a given string.

```
function removeExclamationMarks(s) {
  let result = []
  let array = s.split('')
  for(let i = 0; i < array.length; i++){
    if(array[i] !== '!'){
      result.push(array[i])
    }
  }
  return result.join('');
}
```
